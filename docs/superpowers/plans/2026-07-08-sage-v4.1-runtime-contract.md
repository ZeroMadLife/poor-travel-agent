# Sage v4.1 Runtime Contract Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the first Hermes-style runtime-contract layer for Sage: typed run events, a dedicated ToolExecutor boundary, and frontend stream/event decomposition without changing the visible product flow.

**Architecture:** Backend events become JSON-safe Pydantic models constructed from `core/coding/events.py`. `Engine` keeps model-loop control while `ToolExecutor` owns tool payload normalization, permission/policy checks, approval waiting, tool execution, and tool events. Frontend keeps the current Vue UI but extracts event reduction and WebSocket lifecycle out of the large Pinia store.

**Tech Stack:** Python 3.11+, FastAPI, Pydantic v2, pytest, Vue 3, Pinia, TypeScript, Vitest.

---

## File Structure

- Create `core/coding/events.py`: typed run-event models and helper constructors.
- Create `core/coding/tool_executor.py`: tool execution boundary moved out of `Engine`.
- Modify `core/coding/engine.py`: accept `run_id`, instantiate/use `ToolExecutor`, convert typed events to existing JSON dicts at yield boundary.
- Modify `core/coding/runtime.py`: pass `run_id` into `Engine`; avoid double-stamping event ids when events already have them.
- Modify `core/coding/run_store.py`: keep accepting dicts, add tests proving typed event dicts still summarize correctly.
- Create `tests/core/coding/test_events.py`: event model serialization and required-field coverage.
- Create `tests/core/coding/test_tool_executor.py`: ToolExecutor behavior coverage.
- Modify `tests/core/coding/test_engine.py`: account for `run_id` / `created_at` while keeping existing event order.
- Modify `tests/api/test_coding_routes.py`: assert WebSocket event compatibility with extra `run_id` / `created_at`.
- Create `frontend/src/stores/codingEvents.ts`: pure-ish event reducer helpers.
- Create `frontend/src/stores/codingEvents.test.ts`: event reducer tests.
- Create `frontend/src/stores/codingStream.ts`: WebSocket lifecycle wrapper.
- Create `frontend/src/stores/codingStream.test.ts`: stream controller tests.
- Modify `frontend/src/stores/coding.ts`: delegate event handling and socket lifecycle to new modules; keep public store API stable.
- Modify `frontend/src/types/api.ts`: add optional `run_id` / `created_at` to coding events without breaking existing tests.

Do not modify `agents/`, `mcp_servers/`, `core/verifier.py`, or `evals/`.

```python
"""Typed run events for the Sage coding runtime."""

from __future__ import annotations

from typing import Any, Literal, TypeAlias

from pydantic import BaseModel, Field

from core.coding.workspace import now


class RunEventBase(BaseModel):
    """Base metadata every persisted run event carries."""

    type: str
    run_id: str
    created_at: str = Field(default_factory=now)


class TurnStartedEvent(RunEventBase):
    """A runtime turn has started."""

    type: Literal["turn_started"] = "turn_started"


class TurnFinishedEvent(RunEventBase):
    """A runtime turn has finished."""

    type: Literal["turn_finished"] = "turn_finished"


class ModelRequestedEvent(RunEventBase):
    """The engine is about to call the model."""

    type: Literal["model_requested"] = "model_requested"
    attempts: int
    tool_steps: int
    prompt_chars: int


class ModelParsedEvent(RunEventBase):
    """The engine parsed one model response."""

    type: Literal["model_parsed"] = "model_parsed"
    kind: str


class ToolCallEvent(RunEventBase):
    """A tool is about to execute."""

    type: Literal["tool_call"] = "tool_call"
    tool: str
    args: dict[str, Any] = Field(default_factory=dict)


class ApprovalRequiredEvent(RunEventBase):
    """A tool execution is blocked on user approval."""

    type: Literal["approval_required"] = "approval_required"
    approval_id: str
    tool: str
    args: dict[str, Any] = Field(default_factory=dict)
    description: str
    pattern_key: str


class ApprovalGrantedEvent(RunEventBase):
    """A pending approval was granted."""

    type: Literal["approval_granted"] = "approval_granted"
    tool: str


class ToolResultEvent(RunEventBase):
    """A tool completed, failed, or was denied."""

    type: Literal["tool_result"] = "tool_result"
    tool: str
    args: dict[str, Any] = Field(default_factory=dict)
    content: str
    is_error: bool = False
    policy_reason: str | None = None
    security_event_type: str | None = None


class RetryEvent(RunEventBase):
    """The model output failed the protocol and needs retry."""

    type: Literal["retry"] = "retry"
    content: str


class FinalEvent(RunEventBase):
    """The assistant produced its final answer."""

    type: Literal["final"] = "final"
    content: str


class StepLimitEvent(RunEventBase):
    """The engine stopped after the max tool-step budget."""

    type: Literal["step_limit"] = "step_limit"
    content: str


class CancelledEvent(RunEventBase):
    """The current run was cancelled."""

    type: Literal["cancelled"] = "cancelled"
    content: str


class ErrorEvent(RunEventBase):
    """The runtime surfaced an error."""

    type: Literal["error"] = "error"
    message: str


RunEvent: TypeAlias = (
    TurnStartedEvent
    | TurnFinishedEvent
    | ModelRequestedEvent
    | ModelParsedEvent
    | ToolCallEvent
    | ApprovalRequiredEvent
    | ApprovalGrantedEvent
    | ToolResultEvent
    | RetryEvent
    | FinalEvent
    | StepLimitEvent
    | CancelledEvent
    | ErrorEvent
)


def event_to_dict(event: RunEvent | dict[str, Any]) -> dict[str, Any]:
    """Return a JSON-safe dict for WebSocket, run store, and tests."""
    if isinstance(event, dict):
        return dict(event)
    return event.model_dump(exclude_none=True)
```

```python
"""Tool execution boundary for the Sage coding engine."""

from __future__ import annotations

import asyncio
from collections.abc import AsyncIterator, Callable
from typing import Any

from core.coding.approval import ApprovalManager, check_dangerous_command
from core.coding.engine_helpers import normalize_tool_payload
from core.coding.events import (
    ApprovalGrantedEvent,
    ApprovalRequiredEvent,
    CancelledEvent,
    RunEvent,
    ToolCallEvent,
    ToolResultEvent,
)
from core.coding.permissions import PermissionChecker
from core.coding.tool_policy import ToolPolicyChecker
from core.coding.tools.base import RegisteredTool, ToolResult
from core.coding.workspace import WorkspaceContext


class ToolExecutor:
    """Execute one model-requested tool payload and emit typed events."""

    def __init__(
        self,
        *,
        run_id: str,
        session_id: str,
        workspace: WorkspaceContext,
        tools: dict[str, RegisteredTool],
        permission_checker: PermissionChecker,
        policy_checker: ToolPolicyChecker,
        approval_manager: ApprovalManager | None,
        should_stop: Callable[[], bool],
    ) -> None:
        self.run_id = run_id
        self.session_id = session_id
        self.workspace = workspace
        self.tools = tools
        self.permission_checker = permission_checker
        self.policy_checker = policy_checker
        self.approval_manager = approval_manager
        self.should_stop = should_stop

    async def execute(self, payload: Any) -> AsyncIterator[RunEvent]:
        if self.should_stop():
            yield self._cancelled_event()
            return
        name, args = normalize_tool_payload(payload)
        tool = self.tools.get(name)
        if tool is None:
            yield ToolResultEvent(
                run_id=self.run_id,
                tool=name,
                args=args,
                content=f"unknown tool: {name}",
                is_error=True,
            )
            return

        permission = self.permission_checker.check(tool, args, self.workspace)
        if not permission.allowed:
            yield ToolResultEvent(
                run_id=self.run_id,
                tool=name,
                args=args,
                content=permission.reason,
                is_error=True,
                security_event_type=permission.security_event_type,
            )
            return

        policy = self.policy_checker.check(tool, args)
        if not policy.allowed:
            yield ToolResultEvent(
                run_id=self.run_id,
                tool=name,
                args=args,
                content=policy.message,
                is_error=True,
                policy_reason=policy.reason,
            )
            return

        if permission.reason == "approval_required":
            approved = False
            async for event in self._approval_events(tool, args):
                if event.type == "approval_granted":
                    approved = True
                    continue
                yield event
                if event.type in {"tool_result", "cancelled"}:
                    return
            if not approved:
                return

        yield ToolCallEvent(run_id=self.run_id, tool=name, args=args)
        result = tool.execute(args)
        if self.should_stop():
            yield self._cancelled_event()
            return
        yield ToolResultEvent(
            run_id=self.run_id,
            tool=name,
            args=args,
            content=result.content,
            is_error=result.is_error,
        )

    async def _approval_events(
        self,
        tool: RegisteredTool,
        args: dict[str, Any],
    ) -> AsyncIterator[RunEvent]:
        if self.approval_manager is None or not self.session_id:
            yield ToolResultEvent(
                run_id=self.run_id,
                tool=tool.name,
                args=args,
                content="approval manager is not configured",
                is_error=True,
            )
            return

        description = f"{tool.name} requires approval."
        pattern_key = f"tool:{tool.name}"
        if tool.name == "run_shell":
            dangerous, command_description, command_pattern = check_dangerous_command(
                str(args.get("command", ""))
            )
            if dangerous:
                description = command_description
                pattern_key = f"shell:{command_pattern}"

        if self.approval_manager.is_session_approved(self.session_id, pattern_key):
            yield ApprovalGrantedEvent(run_id=self.run_id, tool=tool.name)
            return

        entry = self.approval_manager.submit(
            self.session_id,
            tool.name,
            args,
            description,
            pattern_key,
        )
        yield ApprovalRequiredEvent(
            run_id=self.run_id,
            approval_id=entry.approval_id,
            tool=tool.name,
            args=args,
            description=description,
            pattern_key=pattern_key,
        )

        waited_seconds = 0
        while waited_seconds < 300:
            if self.should_stop():
                yield ToolResultEvent(
                    run_id=self.run_id,
                    tool=tool.name,
                    args=args,
                    content="approval cancelled",
                    is_error=True,
                )
                yield self._cancelled_event()
                return
            if await asyncio.to_thread(entry.event.wait, 1.0):
                break
            waited_seconds += 1
        if not entry.event.is_set():
            yield ToolResultEvent(
                run_id=self.run_id,
                tool=tool.name,
                args=args,
                content="approval timed out",
                is_error=True,
            )
            return
        if entry.result == "deny":
            yield ToolResultEvent(
                run_id=self.run_id,
                tool=tool.name,
                args=args,
                content="approval denied",
                is_error=True,
            )
            return
        yield ApprovalGrantedEvent(run_id=self.run_id, tool=tool.name)

    def _cancelled_event(self) -> CancelledEvent:
        return CancelledEvent(run_id=self.run_id, content="已停止当前运行。")
```

```typescript
import type {
  CodingApproval,
  CodingApprovalRequiredEvent,
  CodingServerEvent,
  CodingToolCallEvent,
  CodingToolResultEvent,
} from '../types/api'
import type { ChatMessage, ToolActivity } from './coding'

export type CodingEventState = {
  sessionId: string
  messages: ChatMessage[]
  isThinking: boolean
  errorMessage: string
  contextChars: number
  pendingApproval: CodingApproval | null
}

export function createCodingEventState(): CodingEventState {
  return {
    sessionId: '',
    messages: [],
    isThinking: false,
    errorMessage: '',
    contextChars: 0,
    pendingApproval: null,
  }
}

export function applyCodingEvent(state: CodingEventState, event: CodingServerEvent): void {
  if (event.type === 'model_requested') {
    if (event.prompt_chars) state.contextChars = event.prompt_chars
    return
  }
  if (event.type === 'tool_call') {
    appendToolActivity(state, event as CodingToolCallEvent)
    return
  }
  if (event.type === 'approval_required') {
    setApprovalFromEvent(state, event as CodingApprovalRequiredEvent)
    return
  }
  if (event.type === 'tool_result') {
    updateToolActivity(state, event as CodingToolResultEvent)
    return
  }
  if (event.type === 'final' || event.type === 'step_limit' || event.type === 'cancelled') {
    finalizeCurrentMessage(state, event.content)
    state.isThinking = false
    return
  }
  if (event.type === 'error') {
    state.errorMessage = event.message
    state.isThinking = false
  }
}

function appendToolActivity(state: CodingEventState, event: CodingToolCallEvent): void {
  const lastMessage = state.messages[state.messages.length - 1]
  if (!lastMessage || lastMessage.role !== 'assistant' || !lastMessage.isThinking) {
    state.messages.push({
      role: 'assistant',
      content: '',
      tools: [],
      isThinking: true,
    })
  }
  const current = state.messages[state.messages.length - 1]
  current.tools = current.tools || []
  current.tools.push({
    tool: event.tool,
    args: event.args,
    status: 'running',
    content: '',
  } satisfies ToolActivity)
}

function updateToolActivity(state: CodingEventState, event: CodingToolResultEvent): void {
  for (const msg of [...state.messages].reverse()) {
    if (!msg.tools) continue
    const target = [...msg.tools].reverse().find(
      (tool) => tool.tool === event.tool && tool.status === 'running',
    )
    if (target) {
      target.status = event.is_error ? 'error' : 'done'
      target.content = event.content.slice(0, 2000)
      return
    }
  }
}

function setApprovalFromEvent(state: CodingEventState, event: CodingApprovalRequiredEvent): void {
  state.pendingApproval = {
    approval_id: event.approval_id,
    session_id: state.sessionId,
    tool: event.tool,
    args: event.args,
    description: event.description,
    pattern_key: event.pattern_key,
  }
}

function finalizeCurrentMessage(state: CodingEventState, content: string): void {
  const lastMessage = state.messages[state.messages.length - 1]
  if (lastMessage && lastMessage.isThinking) {
    lastMessage.content = content
    lastMessage.isThinking = false
  } else {
    state.messages.push({ role: 'assistant', content })
  }
}
```

```typescript
import type { CodingServerEvent } from '../types/api'

export type CodingStreamOptions = {
  buildUrl: (sessionId: string) => string
  onEvent: (event: CodingServerEvent) => void
  onError: () => void
}

export type CodingStream = {
  connect: (sessionId: string) => void
  send: (payload: { content: string }) => void
  disconnect: () => void
}

export function createCodingStream(options: CodingStreamOptions): CodingStream {
  let socket: WebSocket | null = null

  function connect(sessionId: string): void {
    disconnect()
    socket = new WebSocket(options.buildUrl(sessionId))
    socket.onmessage = (event) => {
      options.onEvent(JSON.parse(event.data) as CodingServerEvent)
    }
    socket.onerror = () => {
      options.onError()
    }
  }

  function send(payload: { content: string }): void {
    if (!socket || socket.readyState !== WebSocket.OPEN) return
    socket.send(JSON.stringify(payload))
  }

  function disconnect(): void {
    socket?.close()
    socket = null
  }

  return { connect, send, disconnect }
}
```
